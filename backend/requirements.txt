# ============================================================================
# NOTES FOR LEARNERS
# ============================================================================
#
# Installing packages:
# 1. Add the package to this file
# 2. Rebuild the Docker container: docker compose up --build
#    OR install in running container: docker compose exec backend pip install -r requirements.txt
#
# Finding package versions:
# - Visit PyPI: https://pypi.org/project/package-name/
# - Check documentation for compatibility
#
# Version specifiers:
# - Django==5.0.1      : Exact version (most strict)
# - Django>=5.0,<5.1   : Allow patch updates (recommended)
# - Django>=5.0        : Allow any newer version (risky)
# - Django~=5.0.1      : Compatible release (>=5.0.1, <5.1.0)
#
# Security updates:
# Run periodically: pip list --outdated
# Update: pip install --upgrade package-name
#
